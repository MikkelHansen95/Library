DROP SCHEMA "libraryschema" CASCADE;
CREATE SCHEMA libraryschema;
CREATE TYPE libraryschema.borrowertype AS ENUM ('Student', 'Teacher', 'Normal');
create table libraryschema.book(
    bookid serial PRIMARY KEY,
    title character varying,
	author character varying,
	publisher character varying,
	isbn character varying(13),
	onLoan boolean,
	location integer,
	publishYear date
);

create table libraryschema.borrower (
    borrowerid serial PRIMARY KEY,
	btype borrowertype,
	bname text,
	address character varying(35),
	cpr integer
);

create table libraryschema.loans (
	bookid serial REFERENCES libraryschema.book,
	borrowerid serial REFERENCES libraryschema.borrower,
	loanID serial PRIMARY KEY,
	startDate date,
	endDate date,
	overdue boolean, 
	activeLoan boolean
);
